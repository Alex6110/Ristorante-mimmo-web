						parent.scrollTo({ left: offset, behavior: 'smooth' });
					}
				} else {
					link.classList.remove('nav-link-active', 'text-[#cfae70]');
				}
			});
				}
			});
			}, { rootMargin: '-30% 0px -60% 0px' });

			subSections.forEach(section => subObserver.observe(section));
		};

		// 3. SBLOCCO PAGINA
		window.addEventListener('load', () => {
			// Piccolo delay per dare tempo a Safari di finire i calcoli di layout
			setTimeout(() => {
			window.scrollTo(0, 0);
			document.documentElement.classList.remove('is-loading');
			
			// Ora che la pagina è ferma in cima, attiviamo lo scroll smooth e gli observer
			document.documentElement.style.scrollBehavior = 'smooth';
			initObservers();
			}, 100);
		});
	</script>
